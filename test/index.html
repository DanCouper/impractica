// Establish root object: `window` in browser, `exports` on server.
var root = this;

var Grid = {};

// Export the Grid object for Node.js, with backwards-compatibility for the old require() API.
// If we're in the browser, add Grid as a global object via a string identifier, for Closure Compiler "advanced" mode.
if (typeof exports !== 'undefined') {
  if (typeof module !== 'undefined' && module.exports) {
    exports = module.exports = Grid;
  }
  exports.Grid = Grid;
} else {
  root.Grid = Grid;
}

Grid.VERSION = '1.6.0';

  else

  end
